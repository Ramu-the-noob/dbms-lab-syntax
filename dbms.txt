Create database : create database database-name;

To use database : use database-name;

To delete database : drop database database-name;

To Show databases : show databases;

To create table : create table t-name(col1 type, col2 type, ...);

To display output : select * from t-name;
                            or
                    select col1,col2,.. from t-name;

To delete table : drop table t-name;

Add column to table : alter table t-name add col_type;

Rename column : alter table t-name rename column col1 to col2;

Modify column : alter table table t-name modify column col type;

Insert data : insert into t-name values(val1, val2, val3, ..);
                            or
              insert into t-name values(row1), (row2), (row3),...;

delete row :

Selecting with condition : select * from t-name where condition;
                           select * from t-name where (condition1 AND,OR,NOT condition2,..);

Updating values : update t-name set col1 = val where condition;

add constraint : alter table add constraint con-name check(condition);

add primary key : create table (col1 type primary key, col2 type,...);

add foreign key : create table (col1 type, col2 type, ... foreign key(col a) references t-name(col a), foreign key(col b) references t-name(col b)..);

joining tables : select * from t-name1 leftjoin t-name2 on t-name1.foreign key = t-name2.foreign key;

                 select * from t-name1 join t-name2 on t-name1.foreign key = t-name2.foreign key;

how to make function pl/sql :

    CREATE OR REPLACE FUNCTION function_name (
    parameter_1 IN data_type,
    parameter_2 IN data_type
)
RETURN return_data_type
IS
    -- Declarative section: Define local variables here
    v_result return_data_type;
BEGIN
    -- Execution block: Your logic goes here
    v_result := parameter_1 + parameter_2;

    RETURN v_result;

EXCEPTION
    -- Exception block: Handle errors gracefully
        DBMS_OUTPUT.PUT_LINE('An error occurred.');
        RETURN NULL;
END;
/

how to make cursor in pl/sql (cursor is temporary area for processing requests):

    DECLARE
    -- 1. Define the cursor with a name and a query
    CURSOR cursor_name IS
        SELECT column1, column2, ...
        FROM table_name
        WHERE condition;
BEGIN

    FOR record_name IN cursor_name LOOP

        -- Access data using: record_name.column_name
        statements_to_execute;

    END LOOP;
END;
/


how to make trigger in pl/sql :

    CREATE OR REPLACE TRIGGER audit_salary_change
BEFORE UPDATE OF salary ON employees
FOR EACH ROW
BEGIN
    -- Insert record into an audit table
    INSERT INTO salary_audit (emp_id, old_salary, new_salary, change_date)
    VALUES (:OLD.employee_id, :OLD.salary, :NEW.salary, SYSDATE);
END;
/


how to make package in pl/sql :

    Head

    CREATE [OR REPLACE] PACKAGE package_name IS

    -- Public Constant Declarations
    constant_name CONSTANT data_type := value;

    -- Public Variable Declarations
    variable_name data_type;

    -- Public Cursor Declarations
    CURSOR cursor_name RETURN return_type;

    -- Public Procedure Declarations (Header only)
    PROCEDURE procedure_name (parameter1 IN data_type, parameter2 OUT data_type);

    -- Public Function Declarations (Header only)
    FUNCTION function_name (parameter1 IN data_type) RETURN return_data_type;

END package_name;
/

    Body

    CREATE OR REPLACE PACKAGE BODY emp_mgmt IS

    -- Private function (cannot be called from outside the package)
    FUNCTION calculate_bonus(p_sal NUMBER) RETURN NUMBER IS
    BEGIN
        RETURN p_sal * 0.10;
    END calculate_bonus;

    PROCEDURE add_employee(p_name VARCHAR2, p_sal NUMBER) IS
    BEGIN
        INSERT INTO employees (name, salary) VALUES (p_name, p_sal);
    END add_employee;

    FUNCTION get_annual_salary(p_emp_id NUMBER) RETURN NUMBER IS
        v_sal NUMBER;
    BEGIN
        SELECT salary INTO v_sal FROM employees WHERE employee_id = p_emp_id;
        RETURN (v_sal * 12) + calculate_bonus(v_sal);
    END get_annual_salary;

END emp_mgmt;
/
